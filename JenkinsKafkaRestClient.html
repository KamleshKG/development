Jenkins Kafka REST Client JSON Error Fix
jenkins console output for kafka rest client produce stage event to kafka ...... expected to call groovy.json.JsonOutput.toJson but wound up catching org.jenkinsci.plugins.workflow.cps.CpsClosure2.call; see: https://jenkins.io/redirect/pipeline-cps-method-mismatches/

This error occurs in a Jenkins pipeline when there's a mismatch between the expected method call (groovy.json.JsonOutput.toJson) and what actually gets called (org.jenkinsci.plugins.workflow.cps.CpsClosure2.call). This typically happens in Jenkins Scripted or Declarative Pipelines due to the CPS (Continuation Passing Style) transformation that Jenkins applies to pipeline scripts.

Common Causes and Solutions

  1. Incorrect Closure Usage:
    You're likely trying to use JsonOutput.toJson() inside a pipeline step or closure where it's being transformed by Jenkins' CPS system.

  2. Solution Approaches:

    • Move the JSON serialization outside of pipeline steps

    • Use @NonCPS annotation for methods that do JSON processing

    • Wrap the JSON processing in a separate function

@NonCPS
def postToKafka(String url, String topic, String json, String authToken = null) {
    def fullUrl = "${url}/topics/${topic}"
    def conn = new URL(fullUrl).openConnection() as HttpURLConnection
    conn.requestMethod = 'POST'
    conn.setRequestProperty('Content-Type', 'application/vnd.kafka.json.v2+json')
    if (authToken) conn.setRequestProperty('Authorization', "Bearer ${authToken}")
    conn.doOutput = true

    conn.outputStream.withWriter('UTF-8') { writer ->
        writer.write(json)
    }

    return conn.responseCode == 200 ? conn.inputStream.text : null
}

pipeline {

                                buildId: env.BUILD_ID,
                                event: 'deployment_started'
                            ]
                        ]]
                    ])

                    def response = postToKafka(
                        'http://kafka-rest:8082',
                        'deployments',
                        payload,
                        'your_bearer_token'
                    )

def kafkaProduce(String url, String topic, String username, String password, Map event) {
    // Prepare connection
    def conn = new URL("${url}/topics/${topic}").openConnection() as HttpURLConnection
    conn.requestMethod = "POST"
    conn.doOutput = true
    
    // Headers
    conn.setRequestProperty("Content-Type", "application/vnd.kafka.json.v2+json")
    conn.setRequestProperty("Authorization", "Basic " + "${username}:${password}".bytes.encodeBase64().toString())
    
    // Body
    def payload = groovy.json.JsonOutput.toJson(records: [[value: event]])
    conn.outputStream.withWriter("UTF-8") { it.write(payload) }
    
    // Response handling
    if (conn.responseCode >= 400) {
        error("Kafka produce failed: ${conn.responseCode} - ${conn.errorStream.text}")
    }
    return conn.inputStream.text
}
